Lazy Loading com Retry
app/routes.tsx

Todas as 16+ páginas são lazy-loaded com retry automático em 404 (stale chunks).

function lazyRetry(componentImport) {
  return lazy(async () => {
    const alreadyRetried = sessionStorage.getItem('retry')
    try {
      const component = await componentImport()
      sessionStorage.removeItem('retry')
      return component
    } catch (error) {
      if (!alreadyRetried) {
        sessionStorage.setItem('retry', 'true')
        window.location.reload()
      }
      throw error
    }
  })
}
