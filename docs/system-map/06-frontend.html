lazyRetry() — Stale Chunk Recovery

Wrapper do React.lazy() que intercepta erros 404 de chunks antigos (ex: apos deploy). Usa sessionStorage para evitar loop infinito — tenta reload 1x por pagina por sessao.

function lazyRetry(factory) {
  return lazy(() => factory().catch(() => {
    if (!sessionStorage.getItem(key)) {
      sessionStorage.setItem(key, '1');
      window.location.reload();
    }
    return factory();
  }));
}
